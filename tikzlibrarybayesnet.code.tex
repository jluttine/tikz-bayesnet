\newcommand{\factoredge}[4][]{ %
  % Connect all nodes #2 to all nodes #4 via all factors #3.
  \foreach \f in {#3} { %
    \foreach \x in {#2} { %
      \draw[-,#1] (\x) edge[-] (\f) ;%
    } ;
    \foreach \y in {#4} { %
      \draw[->,#1] (\f) -- (\y) ;%
    } ;
  } ;
  %\draw[<-,#1] (#2) -- (#2-factor) edge[-] (#3) ;
}

\newcommand{\edge}[3][]{ %
  % Connect all nodes #2 to all nodes #3.
  \foreach \x in {#2} { %
    \foreach \y in {#3} { %
      \draw[->,#1] (\x) -- (\y) ;%
    } ;
  } ;
}

\newcommand{\factor}[3][]{ %
  \node[factor, label={[name=#2-caption]#3}, #1] (#2) {} ;  %
}

% name, fitlist, caption
\newcommand{\plate}[3][default]{ %
  \node[wrap=#2] (#1-wrap) {}; %
  \node[plate caption=#1-wrap] (#1-caption) {#3}; %
  \node[plate=(#1-wrap)(#1-caption)] (#1) {}; %
}

% name,  fitlist, input
\newcommand{\gate}[3][default]{ %
  \node[rectangle,draw,dashed, inner sep=2pt, fit=#2] (#1) {}; %
  \foreach \x in {#3} { %
    \draw [-*,thick] (\x) -- (#1); %
  } ;%
}

% name, fitlist-left, caption-left, fitlist-right, caption-right,
% inputs
\newcommand{\vgate}[6]{ %
  % Wrap the left and right parts
  \node[wrap=#2] (#1-left) {}; %
  \node[wrap=#4] (#1-right) {}; %
  % Draw the gate
  \node[gate=(#1-left)(#1-right)] (#1) {}; %
  % Add captions
  \node[caption, below left=of #1.north ] (#1-left-caption)
  {#3}; %
  \node[caption, below right=of #1.north ] (#1-right-caption)
  {#5}; %
  % Draw middle separation
  \draw [-, dashed] (#1.north) -- (#1.south); %
  % Draw inputs
  \foreach \x in {#6} { %
    \draw [-*,thick] (\x) -- (#1); %
  } ;%
}

% name, fitlist-top, caption-top, fitlist-bottom, caption-bottom,
% inputs
\newcommand{\hgate}[6]{ %
  % Wrap the left and right parts
  \node[wrap=#2] (#1-top) {}; %
  \node[wrap=#4] (#1-bottom) {}; %
  % Draw the gate
  \node[gate=(#1-top)(#1-bottom)] (#1) {}; %
  % Add captions
  \node[caption, above right=of #1.west ] (#1-top-caption)
  {#3}; %
  \node[caption, below right=of #1.west ] (#1-bottom-caption)
  {#5}; %
  % Draw middle separation
  \draw [-, dashed] (#1.west) -- (#1.east); %
  % Draw inputs
  \foreach \x in {#6} { %
    \draw [-*,thick] (\x) -- (#1); %
  } ;%
}
